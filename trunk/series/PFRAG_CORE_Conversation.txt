Persona
	requires <*CS_GoalExpo *CORE_INTRO>
	% V%id%01 = One time use counter
	START <if= v%id%01 0 else %1% V= %id%01 1 NewChat Say %id%01 AddChat %id%01>
	result%id%01 <goto %1%>
	Msg%id%01 <I realize that you're just starting out. Do well now and you'll have a bright future as a cavalier.>
	Prompt%id%01 <I'll try my best.>

Persona
	requires <*CS_GoalExpo *CORE_DEF_>
	START <NewChat Say %id%01 AddChat %id%01>
	result%id%01 <goto %1%>
	Msg%id%01 <There have been some troubles in \SCENE RootSceneID lately, I'm sure you're aware...>
	Msg%id%01_1 <>
	CMsg%id%01_1 <ifNPCOK &EnemyNPC Accept>
	Msg%id%01_2 <>
	CMsg%id%01_2 <ifFactionExists &EnemyFac Accept>
	Prompt%id%01 <That's why I'm here.>
	Prompt%id%01_1 <I know. That's why I came.>
	Prompt%id%01_2 <No kidding.>
