Scene
	name <# Tower>
	requires <*Dungeon .UrbanHellhole>
	special <NOEXIT>
	type <DUNGEON building condemned>
	dentrance <*GoUp>
	content <Some 1 30 Sub *DUNGEON_X>
	mapwidth 30
	mapheight 30
	MonkeyMap
	LockedDoorChance 30
	SecretDoorChance 5

	NeededCells 3

	start <Print 1>
	msg1 <You enter \SCENE SceneID .>

	sub
		Team 1
		SetEnemy 2

		Team 2
		name <Monster>
		type <City Vermin Criminal>
		SetEnemy 1
		Stat 2 1

		room
		width 3
		height 3
		sub
			StairsUp
			Destination -1
		end
	end
