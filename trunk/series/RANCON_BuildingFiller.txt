%% *HOSPITAL_X Content
%%   Things you would find in a hospital... A clinic will usually have one
%%   mandatory doctor content and multiple random HOSPITAL_X contents.
%%   These could be additional doctors, or other things.
Content
	requires <*HOSPITAL_X>
	Content <Required *DOCTOR - Local>

Content
	requires <*HOSPITAL_X>
	minimap <.1.2...3............##=##>
	ZONE_SPECIAL <NoGO>
	element1 <Prefab>
	element2 <Prefab>
	element3 <Prefab>
	get%1% <if= L%id%01 0 L= %id%01 1 AddLawful -3>
	get%2% <if= L%id%01 0 L= %id%01 1 AddLawful -3>
	get%3% <if= L%id%01 0 L= %id%01 1 AddLawful -3>
	sub
		STC OneShotDoor
	end
	inv
		Food 0
		Name <Quick Fix Pill>
		FoodQuantity 5
		Effect <2 10 20 0>
		FoodXValue 1000
		desc <This medicine promotes rapid cellular regeneration, and can instantly heal many injuries.>

		Food 0
		Name <Regen Boost Pill>
		FoodQuantity 5
		Effect <1 12 3 25>
		FoodXValue 1500
		desc <This medicine increases the body's ability to repair itself. While it is in effect injuries will be healed much faster than normal.>

		FirstAidFuel 120
		Name <Medicated Bandages>
		desc <Premedicated sterile bandages.>
		Mass 1
	end
