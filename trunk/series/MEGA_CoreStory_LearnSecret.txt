%%
%% *CS_LearnSecretAboutNPC Content
%%
%% The PC will have a chance to learn something about this particular NPC.
%% If this subplot is won, the following trigger must be set:
%%		.%id%_%plotid%_GoLearnSecret
%%
%% PARAM1: The Secret
%% PARAM2: The NPC
%%

Content
	name <Drinking Buddy Knows All>
	desc <You can learn something from E2's drinking buddy.>
	requires <*CS_LearnSecretAboutNPC>

	% E1 is the secret
	% E2 is the NPC whom it involves
	% E3 is a public scene
	% E4 is the buddy
	Element3 <Scene Public !Near -7>
	Element4 <Character ADVEN -Ally !Near -7 !Okay 2 NeverFail>
	Place4 <3 sd ally>

	start <ifNPCDead %4% LoseSubPlot %plotid%>


	sub
		Persona 4
		rumor%id% <%name4% is %name1%'s drinking buddy.>
		% For now, just a trivial one
		greeting <if= PlotStatus %plotid% %id% else GoChat NewChat Say 1 Trigger0 .%id%_%plotid%_GoLearnSecret WinSubPlot %plotid%>
		GoChat <NewChat SayAnything>
		Msg1 <You know what? \SECRET %1%>
	end
